Purpose: This script will broadcast the message "Die Prims" across the region on the objectChannel = -9. This will cause any objects with the PrimDeleter script to get deleted.
Use: This should be applied to a button. I called mine DeleterButton.
*/
integer objectChannel = -9;

default
{
    touch_start(integer num)
    {
        llRegionSay(objectChannel, "Die Prims"); 
    }
}
